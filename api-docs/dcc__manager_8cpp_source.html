123void DCCManager::restart() {
124 // Stop processing if currently disabled
125 if (!enabled) {
126 if (initialized) {
127 LOG_INFO(TAG, "Stopping DCC decoder");
128 initialized = false;
129 }
130 return;
131 }
132
133 LOG_INFOF(TAG, "Starting DCC decoder on pin %d with address %d", pin, address);
134
135 // Configure GPIO pin for DCC signal input with pull-up resistor
136 pinMode(pin, INPUT_PULLUP);
137
138 // Initialize NmraDcc library with manufacturer and version information
139 dcc.pin(0, pin, 0); // Pin 0 = DCC signal input, pin 1 = not used (0)
140 dcc.init(MAN_ID_DIY, 1, FLAGS_AUTO_FACTORY_DEFAULT, 0);
141
142 // Validate DCC address range for accessory decoders
143 if (address >= 1 && address <= 2048) {
144 // NmraDcc library expects 0-based addressing for accessories
145 LOG_INFOF(TAG, "Setting DCC accessory address to %d", address);
146 initialized = true;
147 } else {
148 LOG_ERRORF(TAG, "Invalid DCC address: %d (must be 1-2048)", address);
149 return;
150 }
151
152 LOG_INFO(TAG, "DCC decoder started successfully");
153}
