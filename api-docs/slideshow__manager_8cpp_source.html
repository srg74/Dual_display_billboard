126void SlideshowManager::updateSlideshow() {
127 if (!slideshowActive || enabledImages.empty() || !settingsManager) {
128 return;
129 }
130
131 unsigned long currentTime = millis();
132 uint32_t imageInterval = settingsManager->getImageInterval() * 1000; // Convert seconds to ms
133
134 if (currentTime - lastImageChange >= imageInterval) {
135 // Check if clock is enabled and we've shown all images
136 bool clockEnabled = settingsManager->isClockEnabled();
137
138 if (clockEnabled && !showingClock && currentImageIndex == enabledImages.size() - 1) {
139 // Time to show clock after last image
140 showingClock = true;
141 showClock();
142 } else if (showingClock) {
143 // Clock was showing, now return to first image
144 showingClock = false;
145 currentImageIndex = 0;
146 showNextImage();
147 } else {
148 // Normal image progression
149 currentImageIndex = (currentImageIndex + 1) % enabledImages.size();
150 showNextImage();
151 }
152
153 lastImageChange = currentTime;
154 }
155}

244bool SlideshowManager::shouldRetrySlideshow() const {
245 // If slideshow is active, no need to retry
246 if (slideshowActive) {
247 return false;
248 }
249
250 // If we've never checked for no images, allow retry
251 if (lastNoImagesCheck == 0) {
252 return true;
253 }
254
255 // Check if enough time has passed since last "no images" check
256 // Use the same interval as image display (default 30 seconds, configurable to 10)
257 unsigned long currentTime = millis();
258 uint32_t checkInterval = SLIDESHOW_DEFAULT_INTERVAL_MS; // 10 seconds as requested
259 if (settingsManager) {
260 checkInterval = settingsManager->getImageInterval() * 1000; // Convert to ms
261 // Minimum 10 seconds to avoid rapid loops
262 if (checkInterval < SLIDESHOW_DEFAULT_INTERVAL_MS) {
263 checkInterval = SLIDESHOW_DEFAULT_INTERVAL_MS;
264 }
265 }
266
267 return (currentTime - lastNoImagesCheck >= checkInterval);
268}

300void SlideshowManager::loadEnabledImages() {

302
303 // Get list of all images from the images directory

305 if (!dir || !dir.isDirectory()) {

313 String filename = file.name();

322 }

363 Serial.printf("=== UPDATING IMAGE STATE ===\n");

401void SlideshowManager::loadImageStatesFromStorage() {
402 Serial.println("Loading image enabled states from storage...");
403
404 // Create default file if it doesn't exist to prevent VFS errors
405 if (!LittleFS.exists("/slideshow_states.json")) {
406 File file = LittleFS.open("/slideshow_states.json", "w");
407 if (file) {
408 file.print("{}"); // Empty JSON object
409 file.close();
410 }
411 }
412
413 File file = LittleFS.open("/slideshow_states.json", "r");
414 if (!file) {
415 Serial.println("No stored image states found - all images will be enabled by default");
416 return;
417 }
418
419 String content = file.readString();
420 file.close();
421
422 // Simple JSON parsing for states
423 // Format: {"image1.jpg":true,"image2.jpg":false}
424 content.trim();
425 if (content.startsWith("{") && content.endsWith("}")) {
426 content = content.substring(1, content.length() - 1); // Remove { }
427
428 int pos = 0;
429 while (pos < content.length()) {
430 int colonPos = content.indexOf(':', pos);
431 int commaPos = content.indexOf(',', pos);
432 if (commaPos == -1) commaPos = content.length();
433
434 if (colonPos > pos && colonPos < commaPos) {
435 String filename = content.substring(pos, colonPos);
436 String enabled = content.substring(colonPos + 1, commaPos);
437
438 // Clean quotes
439 filename.replace("\"", "");
440 enabled.trim();
441
442 imageEnabledStates[filename] = (enabled == "true");
443 Serial.printf("Loaded state: %s = %s\n", filename.c_str(), enabled.c_str());
444 }
445
446 pos = commaPos + 1;
447 }
448 }
449
450 Serial.printf("Loaded %d image states from storage\n", imageEnabledStates.size());
451}
