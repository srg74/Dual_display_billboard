122 LOG_ERROR(TAG, "Failed to open credentials file for reading");

125
126 // Read entire file content
127 String content = file.readString();
128 file.close();
129
130 LOG_DEBUGF(TAG, "Credentials file content: %s", content.c_str());
131
132 // Parse JSON content with improved boundary detection
133 int ssidStart = content.indexOf("\"ssid\":\"") + 8;
134 int ssidEnd = content.indexOf("\",\"password\"");
135 int passStart = content.indexOf("\"password\":\"") + 12;
136 int passEnd = content.indexOf("\"}");
137
138 // Validate JSON structure and extract credentials
139 if (ssidStart > 7 && ssidEnd > ssidStart && passStart > 11 && passEnd > passStart) {
140 creds.ssid = content.substring(ssidStart, ssidEnd);
141 creds.password = content.substring(passStart, passEnd);
142
143 // Unescape JSON special characters
144 creds.ssid.replace("\\\"", "\"");
145 creds.password.replace("\\\"", "\"");
146
147 // Mark credentials as valid if SSID is non-empty
148 creds.isValid = (creds.ssid.length() > 0);
149
150 LOG_INFOF(TAG, "Loaded credentials for SSID: %s", creds.ssid.c_str());
151 } else {
