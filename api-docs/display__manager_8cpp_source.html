88void DisplayManager::initializeBacklight() {
89 LOG_INFO("DISPLAY", "Setting up backlights...");
90
91#ifdef ESP32S3_MODE
92 // ESP32S3: Use higher resolution PWM for better brightness control
93 // Different pins but optimized PWM settings to match ESP32 brightness levels
94 // Note: ESP32S3 also has text rendering differences requiring position adjustments
95 // in TextUtils and DisplayClockManager for consistent Unicode character display
96 ledcAttachPin(TFT_BACKLIGHT1_PIN, 3); // GPIO 7 → Channel 3
97 ledcSetup(3, 5000, 10); // Channel 3, 5 KHz, 10-bit (higher resolution than ESP32)
98 ledcWrite(3, 1023); // Full brightness (10-bit max = 1023)
99
100 ledcAttachPin(TFT_BACKLIGHT2_PIN, 4); // GPIO 8 → Channel 4
101 ledcSetup(4, 5000, 10); // Channel 4, 5 KHz, 10-bit (higher resolution than ESP32)
102 ledcWrite(4, 1023); // Full brightness (10-bit max = 1023)
103
104 LOG_INFO("DISPLAY", "ESP32S3 backlights initialized (PWM channels 3,4, 10-bit, 5kHz - optimized for brightness)");
105#else
106 // ESP32 original setup (working) - keep PWM
107 ledcAttachPin(TFT_BACKLIGHT1_PIN, 1); // GPIO 22 → Channel 1
108 ledcSetup(1, 5000, 8); // Channel 1, 5 KHz, 8-bit
109 ledcWrite(1, 255); // Full brightness
110
111 ledcAttachPin(TFT_BACKLIGHT2_PIN, 2); // GPIO 27 → Channel 2
112 ledcSetup(2, 5000, 8); // Channel 2, 5 KHz, 8-bit
