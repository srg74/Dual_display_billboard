226bool ImageManager::saveImage(const String& filename, uint8_t* data, size_t length) {
227 String filepath = String(IMAGES_DIR) + "/" + filename;
228
229 File file = LittleFS.open(filepath, "w");
230 if (!file) {
231 return false;
232 }
233
234 size_t written = file.write(data, length);
235 file.close();
236
237 if (written != length) {
238 LittleFS.remove(filepath);
239 return false;
240 }
241
242 // Verify file was saved
243 if (!LittleFS.exists(filepath)) {
244 return false;
245 }
246
247 // Create image info
248 ImageInfo info;
249 info.filename = filename;
250 info.uploadTime = String(millis()); // Simple timestamp
251 info.fileSize = length;
252 info.isValid = true;
253
254 // Get dimensions
255 // Extract image dimensions if possible
256 uint16_t w, h;
257 if (TJpgDec.getJpgSize(&w, &h, data, length) == JDR_OK) {
258 info.width = w;
259 info.height = h;
260 }
261
262 // Update persistent metadata
263 saveImageInfo(info);
264 updateImageList();
265
266 return true;
267}

283bool ImageManager::handleImageUpload(const String& filename, uint8_t* data, size_t length) {
284 // Enforce maximum image count for memory management
285 uint32_t currentCount = getImageCount();
286 if (currentCount >= MAX_IMAGE_COUNT) {
287 lastErrorMessage = "Maximum image limit reached (" + String(currentCount) + "/" + String(MAX_IMAGE_COUNT) + " images). Please delete some images first.";
288 return false;
289 }
290
291 // Comprehensive JPEG validation and dimension checking
292 if (!validateImageFile(filename, data, length)) {
293 lastErrorMessage = "Image validation failed - invalid format or dimensions";
294 return false;
295 }
296
297 // Check storage space
298 if (!isStorageAvailable(length)) {
299 lastErrorMessage = "Insufficient storage space available";
300 return false;
301 }
302
303 // Save the image
304 bool saveResult = saveImage(filename, data, length);
305 if (!saveResult) {
306 lastErrorMessage = "Failed to save image to storage";
307 } else {
308 lastErrorMessage = ""; // Clear error on success
309 }

333bool ImageManager::displayImage(const String& filename, uint8_t displayNum) {
334 String filepath = String(IMAGES_DIR) + "/" + filename;
335
336 if (!LittleFS.exists(filepath)) {
337 // Log message removed for compilation
338 return false;
339 }
340
341 File file = LittleFS.open(filepath, "r");
342 if (!file) {
343 // Log message removed for compilation
344 return false;
345 }
346
347 size_t fileSize = file.size();
348 uint8_t* buffer = (uint8_t*)malloc(fileSize);
349 if (!buffer) {
350 // Log message removed for compilation
351 file.close();
352 return false;
353 }
354
355 file.read(buffer, fileSize);
356 file.close();
357
358 // Set target display for callback
359 currentTargetDisplay = displayNum;
360
361 // Clear the screen before drawing the image
362 if (displayManager) {
363 // Set rotation for all displays and TJpg decoder
364 displayManager->setRotation(0); // Correct rotation for proper orientation
365
366 displayManager->selectDisplayForImage(displayNum); // Use image rotation
367 TFT_eSPI* tft = displayManager->getTFT(displayNum);
368 if (tft) {
369 tft->fillScreen(TFT_BLACK); // Clear screen with black background
370 }
371 // Keep display selected for TJpg drawing - don't deselect yet
372 }
373
374 // Decode and display (display should still be selected from above)
375 bool success = (TJpgDec.drawJpg(0, 0, buffer, fileSize) == JDR_OK);
376
377 // Now deselect the display
378 if (displayManager) {
379 displayManager->deselectAll();
380 }
381
382 free(buffer);
383
384 if (success) {
385 // Log message removed for compilation
386 } else {
387 // Log message removed for compilation
388 }
389
390 return success;
391}

631String ImageManager::getImageListJson() {
632 String json = "{\"images\":[";
633
634 File dir = LittleFS.open(IMAGES_DIR);
635 if (!dir || !dir.isDirectory()) {
636 json += "],\"count\":0,\"error\":\"Directory not found\"}";
637 return json;
638 }
639
640 bool first = true;
641 int fileCount = 0;
642 File file = dir.openNextFile();
643 while (file) {
644 // Yield to prevent blocking web server
645 yield();
646
647 if (!file.isDirectory()) {
648 String filename = file.name();
649
650 // Remove path prefix if present
651 int lastSlash = filename.lastIndexOf('/');
652 if (lastSlash >= 0) {
653 filename = filename.substring(lastSlash + 1);
654 }
655
656 if (filename.endsWith(".jpg") || filename.endsWith(".jpeg")) {
657 if (!first) json += ",";
658
659 json += "{";
660 json += "\"filename\":\"" + filename + "\",";
661 json += "\"size\":" + String(file.size()) + ",";
662 json += "\"valid\":true";
663 json += "}";
664
665 first = false;
666 fileCount++;
667 }
668 }
669 file = dir.openNextFile();
670 }
671
672 json += "],\"count\":" + String(fileCount);
673 json += ",\"displayType\":\"" + getDisplayTypeString() + "\",";
674 json += "\"requiredResolution\":\"" + getRequiredResolution() + "\"}";
675
676 return json;
