130void MemoryManager::update() {
131 // Skip update if monitoring is temporarily disabled
132 if (!monitoringEnabled) {
133 return;
134 }
135
136 unsigned long currentTime = millis();
137
138 // Check if it's time to update
139 if (currentTime - lastMonitorTime < monitorInterval) {
140 return;
141 }
142
143 lastMonitorTime = currentTime;
144 currentStats.uptimeMs = currentTime;
145 currentStats.lastUpdateMs = currentTime;
146
147 // Update all memory subsystems
148 updateHeapStats();
149 updatePsramStats();
150 updateHealthStatus();
151 updateFragmentation();
152
153 // Trigger cleanup if auto-cleanup is enabled and needed
154 if (autoCleanupEnabled) {
155 triggerCleanupIfNeeded();
156 }
157
158 // Record events based on health status
159 if (currentStats.heapHealth >= WARNING) {
160 recordMemoryEvent(HEAP_INTERNAL, currentStats.heapHealth);
161 }
162
163 if (currentStats.psramAvailable && currentStats.psramHealth >= WARNING) {
164 recordMemoryEvent(PSRAM_EXTERNAL, currentStats.psramHealth);
165 }
166}

201void MemoryManager::updateHeapStats() {
202 // SAFETY: Add basic validation for heap corruption without exceptions
203 size_t heapTotal = ESP.getHeapSize();
204 size_t heapFree = ESP.getFreeHeap();
205
206 // Basic sanity check to detect obvious corruption
207 if (heapTotal == 0 || heapFree > heapTotal || heapTotal > 10000000) {
208 Serial.printf("WARNING: Heap corruption detected - total:%d, free:%d\n", heapTotal, heapFree);
209 // Set safe fallback values based on platform
210 #ifdef ESP32S3_MODE
211 currentStats.heapTotal = 500000; // ESP32-S3 typically has more heap
212 currentStats.heapFree = 100000; // Conservative estimate
213 #else
214 currentStats.heapTotal = 300000; // ESP32 classic heap size
215 currentStats.heapFree = 50000; // Conservative estimate
216 #endif
217 currentStats.heapUsed = currentStats.heapTotal - currentStats.heapFree;
218 return;
219 }
220
221 // Values seem reasonable, proceed with normal update
222 currentStats.heapTotal = heapTotal;
223 currentStats.heapFree = heapFree;
224 currentStats.heapUsed = currentStats.heapTotal - currentStats.heapFree;
225 currentStats.heapMinFree = ESP.getMinFreeHeap();
226 currentStats.heapMaxAlloc = ESP.getMaxAllocHeap();
227
228 // Update peak usage tracking
229 if (currentStats.heapUsed > currentStats.peakHeapUsage) {
230 currentStats.peakHeapUsage = currentStats.heapUsed;
231 }
232
233 // Update running average
234 heapSampleCount++;
235 heapSampleSum += currentStats.heapFree;
236 currentStats.avgFreeHeap = heapSampleSum / heapSampleCount;
237
238 // Prevent overflow of sample counting
239 if (heapSampleCount > 1000) {
240 heapSampleCount = heapSampleCount / 2;
241 heapSampleSum = heapSampleSum / 2;
242 }
243}

344void MemoryManager::updateHealthStatus() {
345 // Calculate health status for each subsystem

357 currentStats.overallHealth = currentStats.psramHealth;

388void MemoryManager::updateFragmentation() {
389 // Calculate heap fragmentation
390 if (currentStats.heapFree > 0) {
391 float fragmentationRatio = (float)currentStats.heapMaxAlloc / (float)currentStats.heapFree;
392 currentStats.heapFragmentation = (1.0f - fragmentationRatio) * 100.0f;
393
394 // Clamp to valid range
395 if (currentStats.heapFragmentation < 0.0f) currentStats.heapFragmentation = 0.0f;
396 if (currentStats.heapFragmentation > 100.0f) currentStats.heapFragmentation = 100.0f;
397 } else {
398 currentStats.heapFragmentation = 100.0f; // Completely fragmented if no free memory
399 }

433MemoryManager::HealthStatus MemoryManager::calculateHealthStatus(size_t free, size_t total) {
434 if (total == 0) return EXCELLENT; // Not applicable
435
436 float freePercentage = ((float)free / (float)total) * 100.0f;
437
438 if (freePercentage >= 80.0f) return EXCELLENT;
439 if (freePercentage >= 60.0f) return GOOD;
440 if (freePercentage >= 40.0f) return WARNING;
441 if (freePercentage >= 20.0f) return CRITICAL;
442 return EMERGENCY;
443}

754size_t MemoryManager::forceCleanup() {
755 size_t initialFree = ESP.getFreeHeap();
756 size_t freedBytes = 0;
757
758 LOG_INFOF(TAG, "Starting memory cleanup (initial free: %d bytes)", initialFree);
759
760 // 1. Force garbage collection by yielding (non-blocking)
761 yield(); // Single yield instead of loop with delays
762
763 // 2. Clear any cached data (placeholder for future implementation)
764 // This is where we would clear image caches, temporary buffers, etc.
765
766 // 3. Trigger ESP32 heap defragmentation
767 #ifdef ESP_IDF_VERSION
768 heap_caps_check_integrity_all(true);
769 #endif
770
771 // 4. Final yield (non-blocking)
772 yield(); // Single yield instead of loop with delays
773
774 size_t finalFree = ESP.getFreeHeap();
775 if (finalFree > initialFree) {
776 freedBytes = finalFree - initialFree;
777 }
778
779 LOG_INFOF(TAG, "Cleanup completed: freed %d bytes (final free: %d bytes)",
780 freedBytes, finalFree);
781
782 // Update statistics immediately
783 updateHeapStats();
784 updateHealthStatus();
785
786 return freedBytes;
787}
