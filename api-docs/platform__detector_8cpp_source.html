68PlatformDetector::PlatformInfo PlatformDetector::detectPlatform() {
69 PlatformInfo info = {};
70
71 // Get chip information
72 esp_chip_info_t chip_info;
73 esp_chip_info(&chip_info);
74
75 // Detect chip model
76 switch(chip_info.model) {
77 case CHIP_ESP32:
78 info.chipModel = CHIP_ESP32_CLASSIC;
79 strcpy(info.chipName, "ESP32");
80 info.expectedPSRAM = false;
81 break;
82 case CHIP_ESP32S2:
83 info.chipModel = CHIP_ESP32_S2;
84 strcpy(info.chipName, "ESP32-S2");
85 info.expectedPSRAM = false;
86 break;
87 case CHIP_ESP32S3:
88 info.chipModel = CHIP_ESP32_S3;
89 strcpy(info.chipName, "ESP32-S3");
90 info.expectedPSRAM = true; // Most S3 variants have PSRAM
91 break;
92 case CHIP_ESP32C3:
93 info.chipModel = CHIP_ESP32_C3;
94 strcpy(info.chipName, "ESP32-C3");
95 info.expectedPSRAM = false;
96 break;
97 default:
98 info.chipModel = CHIP_UNKNOWN;
99 strcpy(info.chipName, "Unknown");
100 info.expectedPSRAM = false;
101 break;
102 }
103
104 // Detect CPU cores
105 info.cpuCores = chip_info.cores;
106
107 // Detect CPU frequency
108 info.cpuFreqMHz = ESP.getCpuFreqMHz();
109
110 // Detect flash size
111 info.flashSize = ESP.getFlashChipSize();
112
113 // Detect PSRAM availability
114 #ifdef BOARD_HAS_PSRAM
115 info.psramConfigured = true;
116 if (psramFound()) {
117 info.psramActuallyPresent = true;
118 info.psramSize = ESP.getPsramSize();
119 info.psramFree = ESP.getFreePsram();
120
121 // Try to determine PSRAM type based on size
122 if (info.psramSize >= 8 * 1024 * 1024) {
123 strcpy(info.boardVariant, "N16R8 (8MB PSRAM)");
124 } else if (info.psramSize >= 2 * 1024 * 1024) {
125 strcpy(info.boardVariant, "N8R2 (2MB PSRAM)");
126 } else {
127 strcpy(info.boardVariant, "Custom PSRAM");
128 }
129 } else {
130 info.psramActuallyPresent = false;
131 info.psramSize = 0;
132 info.psramFree = 0;
133 strcpy(info.boardVariant, "N8 (No PSRAM)");
134 }
135 #else
136 info.psramConfigured = false;
137 info.psramActuallyPresent = false;
138 info.psramSize = 0;
139 info.psramFree = 0;
140 if (info.chipModel == CHIP_ESP32_CLASSIC) {
141 strcpy(info.boardVariant, "DevKit (No PSRAM)");
142 } else {
143 strcpy(info.boardVariant, "Unknown");
144 }
145 #endif
146
147 // Detect heap information
148 info.heapSize = ESP.getHeapSize();
149 info.heapFree = ESP.getFreeHeap();
150
151 return info;
152}

191void PlatformDetector::printPlatformInfo(const PlatformInfo& info) {
192 LOG_INFOF(TAG, "=== MULTIPLATFORM DETECTION RESULTS ===");
193 LOG_INFOF(TAG, "Chip: %s (%d cores @ %d MHz)", info.chipName, info.cpuCores, info.cpuFreqMHz);
194 LOG_INFOF(TAG, "Board Variant: %s", info.boardVariant);

196 LOG_INFOF(TAG, "Heap: %.1f KB total, %.1f KB free",

198
199 LOG_INFOF(TAG, "PSRAM Configuration: %s", info.psramConfigured ? "ENABLED" : "DISABLED");
200 if (info.psramConfigured) {
201 if (info.psramActuallyPresent) {
202 LOG_INFOF(TAG, "PSRAM Status: DETECTED AND FUNCTIONAL");
203 LOG_INFOF(TAG, "PSRAM: %.1f MB total, %.1f MB free",
204 info.psramSize / (1024.0 * 1024.0), info.psramFree / (1024.0 * 1024.0));
205 } else {
206 LOG_WARNF(TAG, "PSRAM Status: CONFIGURED BUT NOT DETECTED");
207 LOG_WARNF(TAG, "This may indicate hardware mismatch");
208 }
209 } else {
210 LOG_INFOF(TAG, "PSRAM Status: NOT CONFIGURED (ESP32 Classic Mode)");
211 }
212
213 LOG_INFOF(TAG, "Platform Compatibility: %s",
214 (info.expectedPSRAM == info.psramActuallyPresent) ? "OPTIMAL" : "SUBOPTIMAL");
215 LOG_INFOF(TAG, "=====================================");
216}

256bool PlatformDetector::testPSRAMAllocation() {
257 #ifdef BOARD_HAS_PSRAM
258 if (!psramFound()) {
259 LOG_WARNF(TAG, "PSRAM allocation test skipped - no PSRAM detected");
260 return false;
261 }
262
263 LOG_INFOF(TAG, "Testing PSRAM allocation capabilities...");
264
265 // Test different allocation sizes
266 const size_t testSizes[] = {1024, 4096, 16384, 65536, 262144}; // 1KB to 256KB
267 const int numTests = sizeof(testSizes) / sizeof(testSizes[0]);
268 int successCount = 0;
269
270 for (int i = 0; i < numTests; i++) {
271 void* psramPtr = heap_caps_malloc(testSizes[i], MALLOC_CAP_SPIRAM);
272 if (psramPtr != nullptr) {
273 // Test write/read
274 memset(psramPtr, 0xAA, testSizes[i]);
275 bool writeOk = (((uint8_t*)psramPtr)[0] == 0xAA);
276 bool writeOk2 = (((uint8_t*)psramPtr)[testSizes[i]-1] == 0xAA);
277
278 if (writeOk && writeOk2) {
279 LOG_INFOF(TAG, " PSRAM allocation test %d KB: SUCCESS", testSizes[i] / 1024);
280 successCount++;
281 } else {
282 LOG_ERRORF(TAG, " PSRAM allocation test %d KB: WRITE FAILED", testSizes[i] / 1024);
283 }
284
285 heap_caps_free(psramPtr);
286 } else {
287 LOG_ERRORF(TAG, " PSRAM allocation test %d KB: ALLOCATION FAILED", testSizes[i] / 1024);
288 }
289 }
290
291 LOG_INFOF(TAG, "PSRAM allocation test results: %d/%d successful", successCount, numTests);
292 return (successCount == numTests);
293 #else
294 LOG_INFOF(TAG, "PSRAM allocation test skipped - not configured for this platform");
295 return true; // Not an error for non-PSRAM platforms
296 #endif
297}
