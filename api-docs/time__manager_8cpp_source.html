158bool TimeManager::waitForTimeSync(int maxRetries) {
159 LOG_INFO(TAG, "Waiting for time synchronization...");
160
161 unsigned long startTime = micros();
162 const unsigned long attemptInterval = 2000000UL; // 2 seconds in microseconds
163
164 for (int retries = 0; retries < maxRetries; retries++) {
165 time_t currentTime = time(nullptr);
166
167 LOG_INFOF(TAG, "Time sync attempt %d/%d (current time: %ld)", retries + 1, maxRetries, currentTime);
168
169 if (currentTime > 100000) {
170 LOG_INFOF(TAG, "Time synchronized successfully! Current epoch: %ld", currentTime);
171 return true;
172 }
173
174 // Non-blocking wait using micros() - wait 2 seconds between attempts
175 unsigned long attemptStart = micros();
176 while ((micros() - attemptStart) < attemptInterval) {
177 yield(); // Allow other tasks to run
178 }
179 }
180
181 time_t finalTime = time(nullptr);
182 LOG_ERRORF(TAG, "Time sync failed after %d attempts (final time: %ld)", maxRetries, finalTime);
183 return false;
184}
