158bool TimeManager::waitForTimeSync(int maxRetries) {
159 LOG_INFO(TAG, "⏳ Waiting for time synchronization...");
160
161 int retries = 0;
162 time_t currentTime;
163 while ((currentTime = time(nullptr)) < 100000 && retries < maxRetries) {
164 yield(); // Non-blocking yield instead of delay
165 delayMicroseconds(1000000); // 1000ms (1 second) for better sync chances
166 retries++;
167 LOG_INFOF(TAG, "⏳ Time sync attempt %d/%d (current time: %ld)", retries, maxRetries, currentTime);
168 }
169
170 if (currentTime > 100000) {
171 LOG_INFOF(TAG, "✅ Time synchronized successfully! Current epoch: %ld", currentTime);
172 } else {
173 LOG_ERRORF(TAG, "❌ Time sync failed after %d attempts (current time: %ld)", retries, currentTime);
174 }
175
176 return currentTime > 100000;
177}
