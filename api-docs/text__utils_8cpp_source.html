144int TextUtils::drawUnicodeText(TFT_eSPI& tft, const String& text, int x, int y, uint16_t color) {
145 // Platform-specific positioning adjustments for ESP32S3
146 // ESP32S3 has different SPI timing and memory alignment affecting text positioning
147#ifdef ESP32S3_MODE
148 // ESP32S3: Adjust Y position for better umlaut alignment
149 y += 1; // Slight downward shift to compensate for memory alignment differences
150 // ESP32S3: Adjust X position for better character spacing
151 x += 1; // Slight rightward shift for better centering
152#endif
153
154 int currentX = x;
155 int index = 0;
156
157 while (index < text.length()) {
158 int oldIndex = index; // Save index in case we need to detect infinite loops
159 uint16_t codepoint = getNextUTF8Char(text, index);
160
161 // Prevent infinite loops - if index didn't advance, force it
162 if (index == oldIndex) {
163 index++;
164 }
165
166 // If we get 0, only break if we're actually at the end
167 if (codepoint == 0 && index >= text.length()) {
168 break;
169 }
170
171 // Try to find Unicode glyph
172 const UnicodeGlyph* glyph = findGlyph(codepoint);
173 if (glyph) {
174 drawGlyph(tft, glyph, currentX, y, color);
175 int advance = glyph->xAdvance;
176 currentX += advance;
177 } else if (codepoint > 0 && codepoint < 128) {
178 // ASCII character - use built-in font
179 tft.setTextFont(2);
180 tft.setTextColor(color);
181 tft.drawChar(codepoint, currentX, y);
182 int advance = tft.textWidth(String((char)codepoint));
183 currentX += advance;
184 } else if (codepoint > 0) {
185 // Unknown Unicode character - draw placeholder or skip
186 int advance = FONT_WIDTH;
187 currentX += advance;
188 }
189 }
190
191 return currentX - x; // Return total width
192}

197int TextUtils::getUnicodeTextWidth(TFT_eSPI& tft, const String& text) {
198 int totalWidth = 0;
199 int index = 0;
200
201 // Set font to match what we use in rendering
202 tft.setTextFont(2);
203
204 while (index < text.length()) {
205 uint16_t codepoint = getNextUTF8Char(text, index);
206
207 // If we get 0, it means we hit an invalid sequence or end of string
208 if (codepoint == 0) {
209 break;
210 }
211
212 const UnicodeGlyph* glyph = findGlyph(codepoint);
213 if (glyph) {
214 totalWidth += glyph->xAdvance;
215 } else if (codepoint < 128) {
216 // ASCII character - use actual TFT font width measurement
217 totalWidth += tft.textWidth(String((char)codepoint));
218 } else {
219 totalWidth += 6; // Default width for unknown (font 1 standard)
220 }
221 }
222
223 return totalWidth;
224}

229String TextUtils::transliterateText(const String& input) {
230 String result = "";
231 int index = 0;
232
233 while (index < input.length()) {
234 uint16_t codepoint = getNextUTF8Char(input, index);
235
236 // If we get 0, we've hit the end or an invalid sequence
237 if (codepoint == 0) {
238 break;
239 }
240
241 // Check if we have a Unicode glyph for this character
242 const UnicodeGlyph* glyph = findGlyph(codepoint);
243 if (glyph) {
244 // We can render this Unicode character directly
245 result += String((char*)&codepoint); // This might not work correctly for Unicode
246 } else if (codepoint < 128) {
247 // ASCII character
248 result += (char)codepoint;
249 } else {
250 // Character mapping for display (fallback for characters we can't render)
251 switch (codepoint) {
252 case 0x00FC: result += "ue"; break; // ü
253 case 0x00E4: result += "ae"; break; // ä
254 case 0x00F6: result += "oe"; break; // ö
255 case 0x00DC: result += "UE"; break; // Ü

258 case 0x00DF: result += "ss"; break; // ß
259 default:
260 result += "?"; // Unknown character
261 break;
262 }
263 }
264 }

291uint16_t TextUtils::getNextUTF8Char(const String& text, int& index) {
292 if (index >= text.length()) {
293 return 0;
294 }
295
296 uint8_t firstByte = text[index++];
297
298 // ASCII character (0xxxxxxx)
299 if ((firstByte & 0x80) == 0) {
300 return firstByte;
301 }
302
303 // 2-byte UTF-8 (110xxxxx 10xxxxxx)
304 if ((firstByte & 0xE0) == 0xC0) {
305 if (index >= text.length()) {
306 index--; // Back up if we don't have enough bytes
307 return 0;
308 }
309 uint8_t secondByte = text[index++];
310 // Validate second byte format
311 if ((secondByte & 0xC0) != 0x80) {
312 index--; // Back up on invalid sequence
313 return firstByte; // Return first byte as fallback
314 }
315 return ((firstByte & 0x1F) << 6) | (secondByte & 0x3F);
316 }
317
318 // 3-byte UTF-8 (1110xxxx 10xxxxxx 10xxxxxx)
319 if ((firstByte & 0xF0) == 0xE0) {
320 if (index + 1 >= text.length()) {
321 index--; // Back up if we don't have enough bytes
322 return 0;
323 }
324 uint8_t secondByte = text[index++];
325 uint8_t thirdByte = text[index++];
326 // Validate byte formats
327 if ((secondByte & 0xC0) != 0x80 || (thirdByte & 0xC0) != 0x80) {
328 index -= 2; // Back up on invalid sequence
329 return firstByte; // Return first byte as fallback
330 }
331 return ((firstByte & 0x0F) << 12) | ((secondByte & 0x3F) << 6) | (thirdByte & 0x3F);
332 }
333
334 // Invalid or unsupported UTF-8 - return as single byte
335 return firstByte;
336}

341void TextUtils::drawGlyph(TFT_eSPI& tft, const UnicodeGlyph* glyph, int x, int y, uint16_t color) {
342 if (!glyph || !glyph->bitmap) return;
343
344 for (int row = 0; row < glyph->height; row++) {
345 uint8_t rowData = glyph->bitmap[row];
346
347 for (int col = 0; col < glyph->width; col++) {
348 // Check if pixel should be drawn (MSB first)
349 if (rowData & (0x80 >> col)) {
350 tft.drawPixel(x + glyph->xOffset + col, y + glyph->yOffset + row, color);
351 }
352 }
353 }
