458String WiFiManager::scanNetworks() {
459 LOG_INFO(TAG, "Scanning for WiFi networks...");
460
461 // Clear any previous scan results first
462 WiFi.scanDelete();
463
464 // MULTIPLATFORM FIX: For ESP32-S3 in AP mode, temporarily switch to AP+STA for scanning
465 wifi_mode_t currentMode = WiFi.getMode();
466 bool modeChanged = false;
467
468 if (currentMode == WIFI_AP) {
469 LOG_DEBUG(TAG, "Switching to AP+STA mode for scanning");
470 WiFi.mode(WIFI_AP_STA);
471 modeChanged = true;
472 delay(100); // Small delay for mode change
473 }
474
475 // PERFORMANCE FIX: Use synchronous scan for better ESP32-S3 compatibility
476 int n = WiFi.scanNetworks(false, true); // async=false, show_hidden=true
477
478 // Check if scan failed
479 if (n == WIFI_SCAN_FAILED) {
480 LOG_ERROR(TAG, "WiFi scan failed to start");
481 // Restore original mode if changed
482 if (modeChanged) {
483 WiFi.mode(currentMode);
484 }
485 return "[]";
486 }
487
488 // Restore original WiFi mode if we changed it
489 if (modeChanged) {
490 LOG_DEBUG(TAG, "Restoring AP mode after scan");
491 WiFi.mode(currentMode);
492 delay(50); // Small delay for mode change
493 }
494
495 String networks = "[";
496 int validNetworkCount = 0;
497
498 // Process all found networks, but only include valid ones
499 for (int i = 0; i < n; i++) {
500 String ssid = WiFi.SSID(i);
501
502 // FILTER: Skip networks with empty or blank SSIDs
503 if (ssid.length() == 0) {
504 LOG_DEBUGF(TAG, "Skipping network %d with empty SSID", i);
505 continue;
506 }
507
508 // FILTER: Skip whitespace-only SSIDs
509 ssid.trim(); // Trim in place (modifies the string)
510 if (ssid.length() == 0) {
511 LOG_DEBUGF(TAG, "Skipping network %d with whitespace-only SSID", i);
512 continue;
513 }
514
515 // FILTER: Skip very weak signals (optional - less than -90 dBm)
516 if (WiFi.RSSI(i) < -90) {
517 LOG_DEBUGF(TAG, "Skipping weak network: %s (%d dBm)", ssid.c_str(), WiFi.RSSI(i));
518 continue;
519 }
520
521 // Add comma separator for valid networks (not first one)
522 if (validNetworkCount > 0) networks += ",";
523
524 // Escape quotes in SSID
525 ssid.replace("\"", "\\\"");
526
527 // Add valid network to JSON
528 networks += "{\"ssid\":\"" + ssid + "\",\"rssi\":" + String(WiFi.RSSI(i)) + "}";
529 validNetworkCount++;
530
531 // Limit to 10 valid networks to prevent memory issues
532 if (validNetworkCount >= 10) {
533 LOG_DEBUGF(TAG, "Reached maximum of 10 networks, stopping scan");
534 break;
535 }
536 }
537
538 networks += "]";
539
540 // Clean up scan results to free memory
541 WiFi.scanDelete();
542
543 LOG_INFOF(TAG, "Found %d total networks, showing %d valid networks", n, validNetworkCount);
544 return networks;
545}

623bool WiFiManager::connectToWiFi(const String& ssid, const String& password) {
624 LOG_INFOF(TAG, "Attempting to connect to WiFi: %s", ssid.c_str());
625
626 // Start the connection
627 WiFi.begin(ssid.c_str(), password.c_str());
628
629 // Wait for connection with timeout (non-blocking in chunks)
630 unsigned long startTime = millis();
631 const unsigned long timeout = 15000; // 15 seconds timeout
632
633 while (WiFi.status() != WL_CONNECTED && millis() - startTime < timeout) {
634 yield(); // Allow other tasks
635 delayMicroseconds(500000); // 500ms delay in microseconds (non-blocking)
636
637 // Log progress every 2 seconds
638 if ((millis() - startTime) % 2000 < 500) {
639 LOG_DEBUGF(TAG, "Connection attempt... Status: %d", WiFi.status());
640 }
641 }
642
643 if (WiFi.status() == WL_CONNECTED) {
644 LOG_INFOF(TAG, "‚úÖ Connected successfully! IP: %s", WiFi.localIP().toString().c_str());
645 LOG_INFOF(TAG, "‚úÖ Signal strength: %d dBm", WiFi.RSSI());
646 return true;
647 } else {
648 LOG_ERRORF(TAG, "‚ùå Connection failed. Status: %d", WiFi.status());
649
650 // Show connection failed status on display

652 displayManager->showQuickStatus("WiFi Failed", TFT_RED);

1995 if (currentMode != MODE_NORMAL) return;
1996
1997 unsigned long currentTime = millis();
1998
1999 // Check connection every 30 seconds
2000 if (currentTime - lastConnectionAttempt < 30000) return;
2001

2005
2006 if (connectionRetryCount <= MAX_RETRY_ATTEMPTS) {

2009 if (currentTime - lastConnectionAttempt >= delay) {
2010 lastConnectionAttempt = currentTime;

2013 LOG_INFO(TAG, "‚úÖ WiFi reconnected successfully");
2014 connectionRetryCount = 0;
2015 } else {
2016 LOG_ERRORF(TAG, "‚ùå Reconnection attempt %d failed", connectionRetryCount);
2017 }
2018 }
2019 } else {
2020 LOG_WARN(TAG, "‚ö†Ô∏è Max retry attempts reached - staying in normal mode, background retries continue");
2021 // Reset counter for background retries every 60 seconds
2022 if (currentTime - lastConnectionAttempt >= 60000) {
2023 connectionRetryCount = 0;
2024 lastConnectionAttempt = currentTime;
2025 }
