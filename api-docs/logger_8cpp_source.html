192void Logger::log(Level level, const String& tag, const String& message) {
193 #if defined(LOGGER_ENABLED) && LOGGER_ENABLED
194 ensureInitialized();
195
196 // Check if this level should be logged based on build flags
197 bool shouldLog = false;
198 switch (level) {
199 case ERROR:
200 shouldLog = LOGGER_LEVEL_ERROR;
201 break;
202 case WARN:
203 shouldLog = LOGGER_LEVEL_WARN;
204 break;
205 case INFO:
206 shouldLog = LOGGER_LEVEL_INFO;
207 break;
208 case DEBUG:
209 shouldLog = LOGGER_LEVEL_DEBUG;
210 break;
211 case VERBOSE:
212 shouldLog = LOGGER_LEVEL_VERBOSE;
213 break;
214 }
215
216 if (!shouldLog || level > currentLevel) return;
217
218 // Get timestamp
219 unsigned long timestamp = millis();
220
221 // Level prefix
222 String prefix;
223 switch (level) {
224 case ERROR: prefix = "[ERROR] "; break;
225 case WARN: prefix = "[WARN] "; break;
226 case INFO: prefix = "[INFO] "; break;
227 case DEBUG: prefix = "[DEBUG] "; break;
228 case VERBOSE: prefix = "[VERBOSE]"; break;
229 }
230
231 // Format: [timestamp] [DEBUG] [TAG] Message
232 String formattedMessage = String("[") + String(timestamp, 10) + "] " +
233 prefix + "[" + tag + "] " + message;
234
235 // Output to Serial
236 Serial.println(formattedMessage);
237
238 // Output to file if file logging is enabled
239 if (fileLogging) {
240 writeToFile(formattedMessage + "\n");
241 }
242 #endif
243}

610bool Logger::enableFileLogging(const String& filePath) {
611 #if defined(LOGGER_ENABLED) && LOGGER_ENABLED
612 ensureInitialized();
613
614 // Initialize LittleFS if not already done
615 if (!LittleFS.begin(true)) {
616 Serial.println("[ERROR] [LOGGER] Failed to initialize LittleFS for file logging");
617 return false;
618 }
619
620 // Create logs directory if it doesn't exist
621 String dirPath = filePath.substring(0, filePath.lastIndexOf('/'));
622 if (dirPath.length() > 0 && !LittleFS.exists(dirPath)) {
623 // Create directory (LittleFS doesn't have mkdir, so we create a temp file and delete it)
624 File tempFile = LittleFS.open(dirPath + "/.temp", "w");
625 if (tempFile) {
626 tempFile.close();
627 LittleFS.remove(dirPath + "/.temp");
628 }
629 }
630
631 // Test if we can write to the file
632 File testFile = LittleFS.open(filePath, "a");
633 if (!testFile) {
634 Serial.println("[ERROR] [LOGGER] Cannot open log file for writing: " + filePath);
635 return false;
636 }
637 testFile.close();
638
639 currentLogFile = filePath;
640 fileLogging = true;
641
642 // Log successful initialization
643 String timestamp = String(millis());
644 String initMessage = "[" + timestamp + "] [INFO] [LOGGER] File logging enabled: " + filePath + "\n";
645 writeToFile(initMessage);
646
647 Serial.println("[INFO] [LOGGER] File logging enabled: " + filePath);
648 return true;
649 #else
650 return false;
651 #endif
652}

713void Logger::writeToFile(const String& message) {
714 #if defined(LOGGER_ENABLED) && LOGGER_ENABLED
715 if (!fileLogging || currentLogFile.length() == 0) {
716 return;
717 }
718
719 File logFile = LittleFS.open(currentLogFile, "a");
720 if (logFile) {
721 logFile.print(message);
722 logFile.close();
723 } else {
724 // If file write fails, disable file logging to prevent spam
725 fileLogging = false;
726 Serial.println("[ERROR] [LOGGER] Failed to write to log file, disabling file logging");
727 }
728 #endif
