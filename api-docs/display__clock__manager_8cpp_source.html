85void DisplayClockManager::displayClockOnDisplay(TFT_eSPI& tft, int csPin) {
86 if (csPin < 0) return;
87
88 // Select target display via chip select
89 digitalWrite(csPin, LOW);
90 tft.setRotation(0); // Portrait orientation for optimal clock display
91 tft.fillScreen(TFT_BLACK); // Clear previous content
92
93 // Render clock label with Unicode support and proper centering
94 tft.setTextColor(TFT_WHITE, TFT_BLACK);
95 String rawLabel = timeManager ? timeManager->getClockLabel() : "Clock";
96 String currentLabel = TextUtils::toDisplayText(rawLabel); // Process international characters
97
98 // Calculate adaptive center position based on display width
99 int textWidth = TextUtils::getUnicodeTextWidth(tft, currentLabel);
100 bool isSmallDisplay = (tft.width() <= 80);
101 int centerX = isSmallDisplay ? 40 : 120; // 80px displays: center=40, 240px displays: center=120
102 int labelY = 20; // Fixed Y position for consistent appearance
103
104 // Draw Unicode text with proper character rendering
105 int labelX = centerX - textWidth / 2;
106 TextUtils::drawUnicodeText(tft, currentLabel, labelX, labelY, TFT_WHITE);
107
108 // Display the selected clock face type
109 switch (currentClockFace) {
110 case CLOCK_CLASSIC_ANALOG:
111 displayAnalogClock(tft);
112 break;
113 case CLOCK_DIGITAL_MODERN:
114 displayDigitalClock(tft);
115 break;
116 case CLOCK_MINIMALIST:
117 displayMinimalistClock(tft);
118 break;
119 case CLOCK_MODERN_SQUARE:
120 displayModernSquareClock(tft);
121 break;
122 default:
123 displayAnalogClock(tft); // Fallback to analog
124 break;
125 }
126
127 digitalWrite(csPin, HIGH);
128}

141void DisplayClockManager::displayAnalogClock(TFT_eSPI& tft) {
142 // Get current time
143 time_t now = time(nullptr);
144 struct tm timeinfo;
145 localtime_r(&now, &timeinfo);
146
147 // Clock face geometry for 80x160 display
148 int centerX = 40, centerY = 80;
149 int radius = 35;
150
151 // Draw outer circle and center dot
152 tft.drawCircle(centerX, centerY, radius, TFT_SKYBLUE);
153 tft.drawCircle(centerX, centerY, 2, TFT_WHITE);
154
155 // Draw hour markers at all 12 positions
156 for (int h = 0; h < 12; h++) {
157 float angle = (h * 30 - 90) * DEG_TO_RAD;
158 int x0 = centerX + cos(angle) * (radius - 8);
159 int y0 = centerY + sin(angle) * (radius - 8);
160 int x1 = centerX + cos(angle) * (radius - 2);
161 int y1 = centerY + sin(angle) * (radius - 2);
162 tft.drawLine(x0, y0, x1, y1, TFT_WHITE);
163 }
164
165 // Calculate hand angles with smooth seconds for minute hand
166 float minAngle = (timeinfo.tm_min * 6 + timeinfo.tm_sec * 0.1 - 90) * DEG_TO_RAD;
167 float hourAngle = ((timeinfo.tm_hour % 12) * 30 + timeinfo.tm_min * 0.5 - 90) * DEG_TO_RAD;
168
169 // Draw hour hand (shorter, red)
170 int hx = centerX + cos(hourAngle) * (radius * 0.5);
171 int hy = centerY + sin(hourAngle) * (radius * 0.5);
172 tft.drawLine(centerX, centerY, hx, hy, TFT_RED);
173
174 // Draw minute hand (longer, red)
175 int mx = centerX + cos(minAngle) * (radius * 0.8);
176 int my = centerY + sin(minAngle) * (radius * 0.8);
177 tft.drawLine(centerX, centerY, mx, my, TFT_RED);
178}

254void DisplayClockManager::displayModernSquareClock(TFT_eSPI& tft) {
255 // Obtain current system time
256 time_t now = time(nullptr);
257 struct tm timeinfo;
258 localtime_r(&now, &timeinfo);
259
260 // Validate time synchronization before rendering
261 if (now < 1000000000) { // Check for reasonable timestamp (post-2001)
262 String clockLabel = getClockLabel();
263 tft.setTextColor(TFT_WHITE, TFT_BLACK);
264 tft.setTextFont(4);
265 int textWidth = tft.textWidth(clockLabel.c_str());
266
267 // Adaptive positioning for different display sizes
268 bool isSmallDisplay = (tft.width() <= 80);
269 int centerX = isSmallDisplay ? 40 : 120;
270 int textX = centerX - textWidth / 2;
271 int textY = isSmallDisplay ? 25 : 30;
272
273 tft.setCursor(textX, textY);
274 tft.print(clockLabel);
275 tft.setCursor(isSmallDisplay ? 5 : 80, isSmallDisplay ? 80 : 120);
276 tft.print("No Time Sync");
277 return;
278 }
279
280 // Calculate display dimensions and positioning
281 bool isSmallDisplay = (tft.width() <= 80);
282 int centerX = isSmallDisplay ? 40 : 120; // Display center X coordinate
283 int centerY = isSmallDisplay ? 80 : 120; // Display center Y coordinate
284 int clockRadius = isSmallDisplay ? 35 : 80; // Clock radius scaled to display
285
286 // Validate calculated coordinates
287 if (centerX < 0 || centerX > tft.width() || centerY < 0 || centerY > tft.height() || clockRadius <= 0) {
288 String clockLabel = getClockLabel();
289 tft.setTextColor(TFT_WHITE, TFT_BLACK);
290 tft.setTextFont(2);
291 int textWidth = tft.textWidth(clockLabel.c_str());
292 int textX = centerX - textWidth / 2;
293 tft.setCursor(textX, isSmallDisplay ? 25 : 30);
294 tft.print(clockLabel);
295 tft.setCursor(isSmallDisplay ? 5 : 80, isSmallDisplay ? 80 : 120);
296 tft.print("Coord Error");
297 return;
298 }
299
300 // Draw rounded square frame with double border for depth
301 tft.drawRoundRect(centerX - clockRadius, centerY - clockRadius,
302 clockRadius * 2, clockRadius * 2, 8, TFT_WHITE);
303 tft.drawRoundRect(centerX - clockRadius + 1, centerY - clockRadius + 1,
304 clockRadius * 2 - 2, clockRadius * 2 - 2, 7, TFT_WHITE);
305
306 // Draw hour markers with emphasis on quarter hours
307 for (int i = 0; i < 12; i++) {
308 float angle = (i * 30 - 90) * PI / 180; // Convert to radians, start from 12 o'clock
309
310 // Quarter-hour markers (12, 3, 6, 9) are longer
311 int markerLength = (i % 3 == 0) ? (isSmallDisplay ? 8 : 12) : (isSmallDisplay ? 4 : 6);
312 int outerX = centerX + (clockRadius - 5) * cos(angle);
313 int outerY = centerY + (clockRadius - 5) * sin(angle);
314 int innerX = centerX + (clockRadius - 5 - markerLength) * cos(angle);
315 int innerY = centerY + (clockRadius - 5 - markerLength) * sin(angle);
316
317 tft.drawLine(innerX, innerY, outerX, outerY, TFT_WHITE);
318 }
319
320 // Calculate hand angles for smooth movement
321 float hourAngle = ((timeinfo.tm_hour % 12) * 30 + timeinfo.tm_min * 0.5 - 90) * PI / 180;
322 float minuteAngle = (timeinfo.tm_min * 6 - 90) * PI / 180;
323
324 // Validate calculated angles
325 if (isnan(hourAngle) || isnan(minuteAngle) || isinf(hourAngle) || isinf(minuteAngle)) {
326 String clockLabel = getClockLabel();
327 tft.setTextColor(TFT_WHITE, TFT_BLACK);
328 tft.setTextFont(2);
329 int textWidth = tft.textWidth(clockLabel.c_str());
330 int textX = centerX - textWidth / 2;
331 tft.setCursor(textX, isSmallDisplay ? 25 : 30);
332 tft.print(clockLabel);
333
334 // Display time as fallback
335 char timeStr[10];
336 snprintf(timeStr, sizeof(timeStr), "%02d:%02d", timeinfo.tm_hour, timeinfo.tm_min);
337 tft.setCursor(isSmallDisplay ? 20 : 80, isSmallDisplay ? 80 : 120);
338 tft.print(timeStr);
339 tft.setCursor(isSmallDisplay ? 5 : 60, isSmallDisplay ? 100 : 140);
340 tft.print("Math Error");
341 return;
342 }
343
344 // Draw minute hand (longer, thin, blue)
345 int minuteHandLength = isSmallDisplay ? 25 : 55;
346 int minuteX = centerX + minuteHandLength * cos(minuteAngle);
347 int minuteY = centerY + minuteHandLength * sin(minuteAngle);
348 tft.drawLine(centerX, centerY, minuteX, minuteY, TFT_BLUE);
349
350 // Draw hour hand (shorter, thick, blue)
351 int hourHandLength = isSmallDisplay ? 18 : 40;
352 int hourX = centerX + hourHandLength * cos(hourAngle);
353 int hourY = centerY + hourHandLength * sin(hourAngle);
354
355 // Make hour hand thicker with parallel line
356 tft.drawLine(centerX, centerY, hourX, hourY, TFT_BLUE);
357 tft.drawLine(centerX + 1, centerY, hourX + 1, hourY, TFT_BLUE);
358
359 // Draw center dot
360 int centerDotRadius = isSmallDisplay ? 3 : 5;
361 tft.fillCircle(centerX, centerY, centerDotRadius, TFT_RED);
362}

413 }
414
415 // Get current time for consistent display across both screens

419
420 // Render on primary display

423 displayManager->selectDisplayForImage(1);
424 displayClockOnDisplay(*tft1, 1);
