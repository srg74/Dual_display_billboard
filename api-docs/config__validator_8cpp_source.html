144void ConfigValidator::loadPlatformConstraints(PlatformType platform) {
145 // Initialize constraints structure with defaults
146 currentConstraints = PlatformConstraints();
147
148 switch (platform) {
149 case PLATFORM_ESP32_DEV:
150 // Standard ESP32 platform constraints
151 currentConstraints.minGPIO = 0;
152 currentConstraints.maxGPIO = 39;
153 currentConstraints.reservedPins = {6, 7, 8, 9, 10, 11}; // Flash SPI pins
154 currentConstraints.inputOnlyPins = {34, 35, 36, 39}; // ADC1 input-only
155 currentConstraints.spiPins = {18, 19, 23}; // VSPI default pins
156 currentConstraints.hasPSRAM = false; // ESP32 classic without PSRAM
157 currentConstraints.maxRAM = 320000; // 320KB internal RAM
158 currentConstraints.platformName = "ESP32-DEV";
159 LOG_INFOF(TAG, "📋 Loaded ESP32-DEV constraints (GPIO 0-39, 320KB RAM)");
160 break;
161
162 case PLATFORM_ESP32_S3:
163 // ESP32-S3 enhanced platform constraints
164 currentConstraints.minGPIO = 0;
165 currentConstraints.maxGPIO = 48;
166 currentConstraints.reservedPins = {26, 27, 28, 29, 30, 31, 32}; // Flash pins
167 currentConstraints.inputOnlyPins = {}; // No input-only restrictions
168 currentConstraints.spiPins = {11, 12, 13}; // SPI2 default pins
169
170 // ESP32-S3 PSRAM detection and memory configuration
171 #ifdef ESP32S3_MODE
172 currentConstraints.hasPSRAM = true; // PSRAM expected in S3 builds
173 currentConstraints.maxRAM = 8192000; // 8MB with PSRAM
174 #else
175 currentConstraints.hasPSRAM = false; // No PSRAM configured
176 currentConstraints.maxRAM = 512000; // 512KB internal RAM only
177 #endif
178
179 currentConstraints.platformName = "ESP32-S3";
180 LOG_INFOF(TAG, "📋 Loaded ESP32-S3 constraints (GPIO 0-48, %s)",
181 currentConstraints.hasPSRAM ? "8MB PSRAM" : "512KB RAM");
182 break;
183
184 default:
185 // Unknown platform - use conservative ESP32 defaults for safety
186 LOG_ERRORF(TAG, "❌ Unknown platform, applying conservative defaults");
187 currentConstraints.minGPIO = 0;
188 currentConstraints.maxGPIO = 39;
189 currentConstraints.reservedPins = {6, 7, 8, 9, 10, 11};
190 currentConstraints.inputOnlyPins = {34, 35, 36, 39};
191 currentConstraints.spiPins = {18, 19, 23};
192 currentConstraints.hasPSRAM = false;
193 currentConstraints.maxRAM = 320000;
194 currentConstraints.platformName = "Unknown";
195 break;
196 }

212ConfigValidator::ValidationReport ConfigValidator::validateSystem(bool autoFix) {
213 LOG_INFOF(TAG, "🔍 Starting comprehensive system validation...");
214
215 // Reset validation report for fresh analysis
216 lastReport = ValidationReport();
217 lastReport.detectedPlatform = detectPlatform();
218 lastReport.detectedDisplay = detectDisplayType();
219
220 // Platform Hardware Validation
221 if (lastReport.detectedPlatform == PLATFORM_UNKNOWN) {
222 addResult(VALIDATION_ERROR, "Platform", "Unknown platform detected",
223 "Check hardware compatibility and ESP-IDF configuration");
224 } else {
225 addResult(VALIDATION_OK, "Platform", "Platform detected: " + getPlatformName());
226 }
227
228 // Display Driver Validation
229 if (lastReport.detectedDisplay == DISPLAY_UNKNOWN) {
230 addResult(VALIDATION_ERROR, "Display", "Unknown display type",
231 "Verify TFT_eSPI display driver configuration");
232 } else {
233 addResult(VALIDATION_OK, "Display", "Display type: " + getDisplayName());
234 }
235
236 // Memory and Resource Validation
237 const auto& stats = MemoryManager::getStats();
238 if (stats.heapTotal > 0) { // Validate memory statistics availability
239 float heapFreePercent = (stats.heapFree * 100.0) / stats.heapTotal;
240 if (heapFreePercent < 20) {
241 addResult(VALIDATION_WARNING, "Memory",
242 String("Low heap memory: ") + String(heapFreePercent, 1) + "%",
243 "Consider reducing memory usage or enabling PSRAM");
244 } else {
245 addResult(VALIDATION_OK, "Memory",
246 String("Memory levels healthy: ") + String(heapFreePercent, 1) + "% free");
247 }
248
249 // PSRAM availability validation for ESP32-S3
250 if (currentConstraints.hasPSRAM) {
251 addResult(VALIDATION_OK, "PSRAM", "PSRAM available and initialized");
252 }
253 } else {
254 addResult(VALIDATION_WARNING, "Memory", "Memory monitoring not available",
255 "MemoryManager may not be initialized");
256 }
257
258 // Execute comprehensive validation across all subsystems
259 validateMemoryConfiguration();
260 validateFileSystemHealth();
261 validateGPIOPin(0, "validation-test");
262
263 // Determine overall system health and readiness
264 lastReport.overallSeverity = calculateOverallSeverity();
265 lastReport.systemReady = (lastReport.fatalCount == 0 && lastReport.errorCount == 0);
266
267 // Log final validation status with appropriate emoji indicators
268 String statusIcon = lastReport.systemReady ? "✅" : "❌";
269 LOG_INFOF(TAG, "%s Validation complete. System Status: %s",
270 statusIcon.c_str(), lastReport.systemReady ? "READY" : "NOT READY");
271
272 if (!lastReport.systemReady) {
273 LOG_WARNF(TAG, "⚠️ System not ready: %d errors, %d fatal issues found",
274 lastReport.errorCount, lastReport.fatalCount);
275 }
276
277 return lastReport;
278}

561bool ConfigValidator::validateGPIOPin(uint8_t pin, const String& purpose) {
562 // Validate pin is within platform-specific GPIO range
563 if (pin < currentConstraints.minGPIO || pin > currentConstraints.maxGPIO) {
564 addResult(VALIDATION_ERROR, "GPIO",
565 String("GPIO ") + String(pin) + " outside valid range (" +
566 String(currentConstraints.minGPIO) + "-" + String(currentConstraints.maxGPIO) + ")",
567 "Use GPIO pin within platform range");
568 return false;
569 }
570
571 // Check for reserved system pins (flash, etc.)
572 for (uint8_t reserved : currentConstraints.reservedPins) {
573 if (pin == reserved) {
574 addResult(VALIDATION_ERROR, "GPIO",
575 String("GPIO ") + String(pin) + " is reserved for system use",
576 "Select different GPIO pin");
577 return false;
578 }
579 }
580
581 // Check input-only restrictions for output purposes
582 if (purpose.indexOf("output") >= 0 || purpose.indexOf("spi") >= 0) {
583 for (uint8_t inputOnly : currentConstraints.inputOnlyPins) {
584 if (pin == inputOnly) {
585 addResult(VALIDATION_ERROR, "GPIO",
586 String("GPIO ") + String(pin) + " is input-only, cannot use for: " + purpose,
587 "Use GPIO pin with output capability");
588 return false;
589 }
590 }
591 }
592
593 addResult(VALIDATION_OK, "GPIO", String("GPIO ") + String(pin) + " valid for: " + purpose);
594 return true;
595}

702bool ConfigValidator::validateFileSystemHealth() {
703 // LittleFS is initialized in main.cpp, so we just need to verify it's working
704 // Attempting to re-mount would create read-only conflicts
705 if (LittleFS.totalBytes() == 0) {
706 addResult(VALIDATION_ERROR, "FileSystem", "LittleFS filesystem not properly initialized",
707 "Check flash partitioning and filesystem integrity");
708 return false;
709 }
710
711 // Check filesystem space availability
712 size_t totalBytes = LittleFS.totalBytes();
713 size_t usedBytes = LittleFS.usedBytes();
714 float usagePercent = (usedBytes * 100.0) / totalBytes;
715
716 if (usagePercent > 90) {
717 addResult(VALIDATION_WARNING, "FileSystem",
718 String("Filesystem nearly full: ") + String(usagePercent, 1) + "% used",
719 "Clean up unnecessary files");
720 } else {
721 addResult(VALIDATION_OK, "FileSystem",
722 String("Filesystem healthy: ") + String(usagePercent, 1) + "% used");
723 }
724
725 return true;
726}
